-- Basic SELECT with filtering
SELECT customer_id, order_date, total_amount
FROM orders
WHERE total_amount > 1000
AND order_date >= '2024-01-01'
ORDER BY total_amount DESC;

-- JOIN operations
SELECT c.customer_name,
COUNT(o.order_id) AS order_count,
SUM(o.total_amount) AS total_spent
FROM customers c
LEFT JOIN orders o ON c.customer_id = o.customer_id
GROUP BY c.customer_id, c.customer_name
HAVING SUM(o.total_amount) > 5000;

-- Window functions
SELECT customer_id,
order_date,
total_amount,
ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY order_date) AS order_sequence,
SUM(total_amount) OVER (PARTITION BY customer_id) AS customer_total
FROM orders;

Common Table Expressions (CTEs)

-- Calculate customer lifetime value with CTEs
WITH customer_metrics AS (
SELECT customer_id,
COUNT(DISTINCT order_id) AS total_orders,
SUM(total_amount) AS total_revenue,
MIN(order_date) AS first_order_date,
MAX(order_date) AS last_order_date
FROM orders
GROUP BY customer_id
),
customer_segments AS (
SELECT *,
CASE
WHEN total_revenue > 10000 THEN 'VIP'
WHEN total_revenue > 5000 THEN 'Gold'
WHEN total_revenue > 1000 THEN 'Silver'
ELSE 'Bronze'
END AS customer_segment
FROM customer_metrics
)
SELECT customer_segment,
COUNT(*) AS customer_count,
AVG(total_revenue) AS avg_revenue
FROM customer_segments
GROUP BY customer_segment;

Choosing the Right Chart

  • Line Chart: Time series data, trends over time
  • Bar Chart: Comparing categories, rankings
  • Scatter Plot: Relationships between two variables
  • Histogram: Distribution of a single variable
  • Heatmap: Correlation matrices, patterns in 2D data

Key Performance Indicators (KPIs)

Common Business Metrics

import pandas as pd
import numpy as np

class BusinessMetrics:
"""Calculate common business KPIs"""

def __init__(self, df):
self.df = df

def conversion_rate(self, conversions_col, visitors_col):
"""Calculate conversion rate"""
conversions = self.df[conversions_col].sum()
visitors = self.df[visitors_col].sum()
return (conversions / visitors) * 100

def customer_churn_rate(self, churned_col, total_col):
"""Calculate churn rate"""
churned = self.df[churned_col].sum()
total = self.df[total_col].sum()
return (churned / total) * 100

def customer_lifetime_value(self, revenue_col, customer_col, periods=12):
"""Calculate CLV"""
avg_purchase_value = self.df[revenue_col].sum() / self.df[customer_col].nunique()
purchase_frequency = len(self.df) / self.df[customer_col].nunique()
return avg_purchase_value * purchase_frequency * periods

def retention_rate(self, cohort_df):
"""Calculate retention by cohort"""
cohort_pivot = cohort_df.pivot_table(
index='cohort_month',
columns='months_since_join',
values='customer_id',
aggfunc='nunique'
)
# Calculate retention percentage
retention = cohort_pivot.divide(cohort_pivot.iloc[:, 0], axis=0) * 100
return retention

def roi(self, revenue_col, cost_col):
"""Calculate Return on Investment"""
total_revenue = self.df[revenue_col].sum()
total_cost = self.df[cost_col].sum()
return ((total_revenue - total_cost) / total_cost) * 100

# Example usage
metrics = BusinessMetrics(sales_df)
print(f"Conversion Rate: {metrics.conversion_rate('purchases', 'visitors'):.2f}%")
print(f"Customer Churn: {metrics.customer_churn_rate('churned', 'total_customers'):.2f}%")
print(f"CLV: ${metrics.customer_lifetime_value('revenue', 'customer_id'):,.2f}")

# E-commerce Metrics
def calculate_ecommerce_metrics(df):
metrics = {}

# Average Order Value (AOV)
metrics['aov'] = df['order_total'].mean()

# Cart Abandonment Rate
carts_created = df['cart_created'].sum()
carts_purchased = df['cart_purchased'].sum()

# Product Return Rate

return metrics

def calculate_saas_metrics(df):
metrics = {}

# Monthly Recurring Revenue (MRR)
metrics['mrr'] = df[df['status'] == 'active']['monthly_fee'].sum()

# Annual Recurring Revenue (ARR)
metrics['arr'] = metrics['mrr'] * 12

# Customer Acquisition Cost (CAC)
marketing_spend = df['marketing_cost'].sum()
new_customers = df[df['customer_type'] == 'new']['customer_id'].nunique()
metrics['cac'] = marketing_spend / new_customers

return metrics

features = ['recency', 'frequency', 'monetary']

# Find optimal number of clusters using elbow method
inertias = []
K_range = range(1, 11)

kmeans.fit(X_scaled)
inertias.append(kmeans.inertia_)

plt.ylabel('Inertia')
plt.title('Elbow Method for Optimal k')
plt.show()

# Apply K-means with optimal k
optimal_k = 4
kmeans = KMeans(n_clusters=optimal_k, random_state=42)
